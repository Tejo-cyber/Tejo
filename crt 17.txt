#include<stdio.h>
void main()
{
    int a=3;
    int b=1;
    printf("%d\n",a++ + --b);//3+0,a=4,b=0
    printf("%d\n",--a + ++b);//3+1=4,a=3,b=1
    printf("%d\n",a++ -6);//3-6=-3,a=4,b=1
    printf("%d\n",--a + b);//3+1=4,a=3,b=1
    printf("%d\n",a - --a);//3-2=0,a=2,b=1
    a=++a + ++b;//3+2=5,a=5,b=2
    b=a++ - --b;//5-1=3,a=6,b=4
    printf("%d\n",a + a - --b);//6+6=12-3=9,a=6,b=3
    printf("%d\n",a - --b + --a);//6-2=4+5=9,a=5,b=2
    printf("%d\n",a++ + a);//5+6=11,a=6,b=2
    printf("%d\t%d\n",++a,b++);//7,2,a=7,b=3
    printf("%d\n",b);//3,a=7,b=3
}
